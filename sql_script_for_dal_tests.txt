The following is the script I run in DBeaver to set up my database for the data access layer tests.

-----
drop table if exists clients cascade; --These two lines are simply to ensure the test environment is built from the ground up so that
drop table if exists accounts cascade; --the return results are what we expect for tests to continue.

create table clients(
	first_name varchar(20),
	last_name varchar(20),
	client_id varchar(20) primary key
);

create table accounts(
	client_id varchar(20),
	account_id varchar(20) primary key,
	account_balance dec(15,2) check(account_balance >= 0),
	constraint accounts_fk foreign key(client_id) references clients(client_id)
);
